433 inline Vector VectorCrossProduct(const Vector & lhs, const Vector & rhs)
434 {
435 bool correctnElements = (lhs.Size() == 3) && (rhs.Size() == 3);
436 if (!correctnElements)
437 {
438 throw std::runtime_error("Cannot compute the cross product for vectors not on R^3");
439 }
440
441 JSL::Vector output(3);
442
443 output[0] = lhs[1]*rhs[2] - rhs[1]*lhs[2];
444 output[1] = lhs[2]*rhs[0] - rhs[2]*lhs[0];
445 output[2] = lhs[0]*rhs[1] - rhs[0]*lhs[1];

447 return output;
