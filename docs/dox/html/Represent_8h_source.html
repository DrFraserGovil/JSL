76 static std::string stringify(const T& container, std::string_view delimiter)
77 {
78 std::ostringstream result;
79 result << "["; // A default endcap
80
81 bool first = true;
82 for (const auto& item : container)
83 {
84 if (!first)
85 {
86 result << delimiter;
87 }
88
89 // Extract the inner type and recurse
90 using InnerType = std::remove_cvref_t<decltype(item)>;
91 result << RepresentStruct<InnerType>::stringify(item);
92
93 first = false;
94 }
95
96 result << "]";
97 return result.str();
98 }
